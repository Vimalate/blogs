## 实现一个函数，入参是一个fn，延迟5s执行，并且拿到返回值

```js
function sleep(fn){
  return new Promise(resolve =>{
    setTimeout(()=>{
      resolve(fn())
    },5000)
  })
}

function f(){
  return '彼时彼刻，恰如此时此刻'
}

console.log(Date.now())
const old = Date.now()
sleep(f).then(res=>{
  console.log(res,Date.now()-old)
})
```
